-	script	::Sailor_izlude	-1,{
	mes "[Моряк]";
	mes "Привет всем!";
	mes "Внимание, внимание!";
	mes "Приходите и прокатитесь с ветерком на нашем прекрасном корабле!";
	mes "Скорее, скорее!";
	next;
	switch(select("Остров Бьялан -> 150 зен.:Альберта Марина -> 500 зен.:Отмена.")) {
	case 1:
		if (Zeny < 150) {
			mes "[Моряк]";
			mes "150 зен!";
			mes "Всего 150 зен, чтобы прокатиться!";
			close;
		}
		set Zeny, Zeny - 150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (Zeny < 500) {
			mes "[Моряк]";
			mes "500 зен!";
			mes "Всего 500 зен для поездки!";
			close;
		}
		set Zeny, Zeny - 500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
